Experiment 5 – Stability Analysis of Linear System using Routh Hurwitz Criterion

Reference Program

coeff=[1 5 6 0];
l=length(coeff);
disp("Roots of Characteristic Equation are:");
disp(roots(coeff));
if modulo(l,2)==0 then
    m=zeros(l,l/2);
else
    m=zeros(l,(l+1)/2);
end
[rows,cols]=size(m);
for i=1:cols
    if (2*i-1)<=l then
        m(1,i)=coeff(2*i-1);
    end
end
for i=1:cols
    if (2*i)<=l then
        m(2,i)=coeff(2*i);
    end
end
for j=3:rows
    if m(j-1,1)==0 then
        m(j-1,1)=0.001;
    end
    for i=1:cols-1
        m(j,i)=((m(j-1,1)*m(j-2,i+1))-(m(j-2,1)*m(j-1,i+1)))/m(j-1,1);
    end
end
disp("Routh-Hurwitz Array:");
disp(m);
Temp=sign(m(:,1));
changes=0;
for i=1:length(Temp)-1
    if Temp(i)*Temp(i+1)<0 then
        changes=changes+1;
    end
end
if changes==0 then
    disp("System is Stable");
else
    disp("System is Unstable");
end
