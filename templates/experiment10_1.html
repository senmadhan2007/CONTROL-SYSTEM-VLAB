Experiment 10 – Design of PID Controller for First Order and Second Order Systems

Reference Program

s = %s;

Sys = syslin('c', 100/(s^3 + 13*s^2 + 32*s + 20));
disp(Sys);

t = 0:0.01:10;

Sys_o = csim('step', t, Sys);

subplot(2,1,1);
plot(t, Sys_o);
xtitle("Open Loop Response of System", "Time (sec)", "Response");

Kp = 1.8;
Ki = 2;
Kd = 0.344;

Con = Kp + Ki/s + Kd*s;
disp(Con);

Sys_f = syslin('c', (Con*Sys) / (1 + Con*Sys));
disp(Sys_f);

Sys_c = csim('step', t, Sys_f);

subplot(2,1,2);
plot(t, Sys_c);
xtitle("Response with PID Controller", "Time (sec)", "Response");
